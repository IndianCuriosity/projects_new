def calculate_numerical_greeks_lv(option_details_dict, mktdata_details_dict, local_vol_surface_base, expiry_years_array, strikes_array, n_paths=100_000, n_steps=252):

    Wrapper framework to calculate Numerical Greeks for Local Volatility Monte Carlo.
    Bumps market data inputs, regenerates or adjusts lookups, and runs finite differences.
    """
    # 1. Calculate Base Price
    print("Calculating Base Price...")
    base_price = VanillaPriceGreeks_MC_Local_Vol(
        option_details_dict, mktdata_details_dict, local_vol_surface_base, 
        expiry_years_array, strikes_array, n_paths, n_steps
    )
    
    # Define standard risk architecture bumps
    h_spot = 0.001 * mktdata_details_dict['Spot'] # 0.1% Spot bump
    h_vega = 0.01                                 # 1% flat vol bump (1 vol point)
    h_time = 1.0 / 365.0                          # 1 day time decay bump
    
    # =========================================================================
    # 🔵 DELTA & GAMMA (Spot Bumps)
    # =========================================================================
    print("\nCalculating Delta & Gamma...")
    # Up Bump
    mkt_up = copy.deepcopy(mktdata_details_dict)
    mkt_up['Spot'] += h_spot
    # In a pure model, you would re-run Dupire if Ft changes. For speed:
    mkt_up['Ft'] = mkt_up['Spot'] * np.exp((mkt_up['rd'] - mkt_up['rf']) * (mkt_up['Te']/365.0))
    price_up = VanillaPriceGreeks_MC_Local_Vol(
        option_details_dict, mkt_up, local_vol_surface_base, 
        expiry_years_array, strikes_array, n_paths, n_steps
    )
    
    # Down Bump
    mkt_down = copy.deepcopy(mktdata_details_dict)
    mkt_down['Spot'] -= h_spot
    mkt_down['Ft'] = mkt_down['Spot'] * np.exp((mkt_down['rd'] - mkt_down['rf']) * (mkt_down['Te']/365.0))

        option_details_dict, mkt_down, local_vol_surface_base, 
        expiry_years_array, strikes_array, n_paths, n_steps
    )
    
    # Finite Difference Formulas
    numerical_delta = (price_up - price_down) / (2 * h_spot)
    numerical_gamma = (price_up - 2 * base_price + price_down) / (h_spot ** 2)
    
    # =========================================================================

    print("\nCalculating Vega...")
    # For Local Vol, Vega requires bumping the underlying implied vols, 
    # re-running Dupire derivatives, and passing the new surface.

    surface_bumped = local_vol_surface_base + h_vega

        option_details_dict, mktdata_details_dict, surface_bumped, 
        expiry_years_array, strikes_array, n_paths, n_steps
    )
    
    # Vega scaled to represent a 1% (1 vol point) parallel volatility move
    numerical_vega = (price_vega_up - base_price) / 1.0 
    
    # =========================================================================
    # 🔴 THETA (Time Decay Bump)

    print("\nCalculating Theta...")
    mkt_theta = copy.deepcopy(mktdata_details_dict)

    )
    
    # Theta expressed as daily decay premium drop
    numerical_theta = (price_theta_forward - base_price) / 1.0

    # =========================================================================

    greeks_results = {

    return greeks_results

from numba import njit

@njit(fastmath=True)
def bilinear_interp_2d(x_grid, y_grid, values, x, y):
    """Blazing fast 2D linear interpolation compiled directly to machine code."""
    # Find binary search slots on the grid (like np.searchsorted)
    i = np.searchsorted(x_grid, x) - 1
    i = max(0, min(i, len(x_grid) - 2))
    
    j = np.searchsorted(y_grid, y) - 1
    j = max(0, min(j, len(y_grid) - 2))
    
    x0, x1 = x_grid[i], x_grid[i+1]
    y0, y1 = y_grid[j], y_grid[j+1]
    
    # Weight factors
    fx = (x - x0) / (x1 - x0)
    fy = (y - y0) / (y1 - y0)
    
    # 4 corners
    v00 = values[i, j]
    v10 = values[i+1, j]
    v01 = values[i, j+1]
    v11 = values[i+1, j+1]
    
    # Interpolate
    return (1 - fx) * (1 - fy) * v00 + fx * (1 - fy) * v10 + (1 - fx) * fy * v01 + fx * fy * v11

@njit(parallel=True, fastmath=True)
def run_numba_mc_loop(Spot, Strike, Te, rd, rf, expiry_years, strikes, lv_surface, n_paths, n_steps, dt, sqrt_dt, shocks):
    """The entire time-stepping loop compiled and executed in parallel."""
    S = np.full(n_paths, Spot)
    half_paths = n_paths // 2
    
    min_strike = strikes[0]
    max_strike = strikes[-1]
    
    for step in range(n_steps):
        t = min(step * dt, expiry_years[-1])
        
        # Pull random normal array values for this step
        Z = shocks[step] 
        
        for p in range(n_paths):
            # Clip spot to prevent extrapolation breakages
            current_S = S[p]
            clipped_S = max(min_strike, min(current_S, max_strike))
            
            # Fast Numba interpolation lookup
            sigma = bilinear_interp_2d(expiry_years, strikes, lv_surface, t, clipped_S)
            if sigma < 1e-5:
                sigma = 1e-5
                
            # Log-Euler evolution step
            S[p] = current_S * np.exp((rd - rf - 0.5 * sigma**2) * dt + sigma * sqrt_dt * Z[p])
            
    return S
